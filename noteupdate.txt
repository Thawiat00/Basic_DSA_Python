2.2 ‡∏™‡∏£‡πâ‡∏≤‡∏á Stack Class (‡πÅ‡∏ö‡∏ö‡∏°‡∏∑‡∏≠‡∏≠‡∏≤‡∏ä‡∏µ‡∏û)
pythonclass Stack:
    def __init__(self):
        self.stack = []
    
    def push(self, element):
        """‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡πÄ‡∏Ç‡πâ‡∏≤ stack"""
        self.stack.append(element)
    
    def pop(self):
        """‡πÄ‡∏≠‡∏≤‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡∏ö‡∏ô‡∏™‡∏∏‡∏î‡∏≠‡∏≠‡∏Å"""
        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"
        return self.stack.pop()
    
    def peek(self):
        """‡∏î‡∏π‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡∏ö‡∏ô‡∏™‡∏∏‡∏î"""
        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"
        return self.stack[-1]
    
    def isEmpty(self):
        """‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤ stack ‡∏ß‡πà‡∏≤‡∏á‡πÑ‡∏´‡∏°"""
        return len(self.stack) == 0
    
    def size(self):
        """‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å"""
        return len(self.stack)

class Stack:
    def __init__(self):
        self.stack = []
    
    def push(self, element):
        """‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡πÄ‡∏Ç‡πâ‡∏≤ stack"""
        self.stack.append(element)
    
    def pop(self):
        """‡πÄ‡∏≠‡∏≤‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡∏ö‡∏ô‡∏™‡∏∏‡∏î‡∏≠‡∏≠‡∏Å"""
        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"
        return self.stack.pop()
    
    def peek(self):
        """‡∏î‡∏π‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡∏ö‡∏ô‡∏™‡∏∏‡∏î"""
        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"
        return self.stack[-1]
    
    def isEmpty(self):
        """‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤ stack ‡∏ß‡πà‡∏≤‡∏á‡πÑ‡∏´‡∏°"""
        return len(self.stack) == 0
    
    def size(self):
        """‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å"""
        return len(self.stack)
