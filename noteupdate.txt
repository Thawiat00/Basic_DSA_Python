        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"
        return self.stack.pop()
    
    def peek(self):
        """‡∏î‡∏π‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡∏ö‡∏ô‡∏™‡∏∏‡∏î"""
        if self.isEmpty():
            return "Stack ‡∏ß‡πà‡∏≤‡∏á"

class Queue:
    def __init__(self):
        self.queue = []
    
    def enqueue(self, element):
        """‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏Ñ‡∏ô‡πÄ‡∏Ç‡πâ‡∏≤‡∏Ñ‡∏¥‡∏ß (‡∏ó‡πâ‡∏≤‡∏¢‡∏™‡∏∏‡∏î)"""
        self.queue.append(element)
    
    def dequeue(self):
        """‡πÄ‡∏≠‡∏≤‡∏Ñ‡∏ô‡∏´‡∏ô‡πâ‡∏≤‡∏™‡∏∏‡∏î‡∏≠‡∏≠‡∏Å"""
        if self.isEmpty():
            return "Queue ‡∏ß‡πà‡∏≤‡∏á"
        return self.queue.pop(0)
    
    def peek(self):
        """‡∏î‡∏π‡∏Ñ‡∏ô‡∏´‡∏ô‡πâ‡∏≤‡∏™‡∏∏‡∏î"""
        if self.isEmpty():
            return "Queue ‡∏ß‡πà‡∏≤‡∏á"
        return self.queue[0]
    
    def isEmpty(self):
        """‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤ queue ‡∏ß‡πà‡∏≤‡∏á‡πÑ‡∏´‡∏°"""
        return len(self.queue) == 0
    
    def size(self):
        """‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏Ñ‡∏ô"""
        return len(self.queue)

def findLowestValue(head):
    minValue = head.data
    currentNode = head.next
    
    while currentNode:
        if currentNode.data < minValue:
            minValue = currentNode.data
        currentNode = currentNode.next
    
    return minValue

    # ‡πÅ‡∏ó‡∏£‡∏Å‡∏ó‡∏µ‡πà‡∏ï‡∏≥‡πÅ‡∏´‡∏ô‡πà‡∏á‡πÅ‡∏£‡∏Å
    if position == 1:
        newNode.next = head
        return newNode
    
    # ‡∏´‡∏≤‡∏ï‡∏≥‡πÅ‡∏´‡∏ô‡πà‡∏á‡∏ó‡∏µ‡πà‡∏à‡∏∞‡πÅ‡∏ó‡∏£‡∏Å
    currentNode = head
    for _ in range(position - 2):
        if currentNode is None:
            break
        currentNode = currentNode.next
    
    # ‡πÅ‡∏ó‡∏£‡∏Å node
    newNode.next = currentNode.next
    currentNode.next = newNode
    return head

# ‡∏ó‡∏î‡∏™‡∏≠‡∏ö
