# ‡∏™‡∏£‡πâ‡∏≤‡∏á Stack
stack = []

# Push (‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÄ‡∏Ç‡πâ‡∏≤ stack)
stack.append('A')
stack.append('B')
stack.append('C')
print("Stack:", stack)

# Peek (‡∏î‡∏π‡∏ï‡∏±‡∏ß‡∏ö‡∏ô‡∏™‡∏∏‡∏î‡πÇ‡∏î‡∏¢‡πÑ‡∏°‡πà‡πÄ‡∏≠‡∏≤‡∏≠‡∏≠‡∏Å)
topElement = stack[-1]
print("Peek:", topElement)

# Pop (‡πÄ‡∏≠‡∏≤‡∏ï‡∏±‡∏ß‡∏ö‡∏ô‡∏™‡∏∏‡∏î‡∏≠‡∏≠‡∏Å)
poppedElement = stack.pop()
print("Pop:", poppedElement)
print("Stack ‡∏´‡∏•‡∏±‡∏á Pop:", stack)

# ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤‡∏á
isEmpty = not bool(stack)
print("Stack ‡∏ß‡πà‡∏≤‡∏á‡πÑ‡∏´‡∏°:", isEmpty)

# ‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô
print("‡∏Ç‡∏ô‡∏≤‡∏î:", len(stack))
